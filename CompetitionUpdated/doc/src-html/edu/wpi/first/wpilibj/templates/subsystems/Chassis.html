001package edu.wpi.first.wpilibj.templates.subsystems;
002
003import edu.wpi.first.wpilibj.CANJaguar;
004import edu.wpi.first.wpilibj.Joystick;
005import edu.wpi.first.wpilibj.RobotDrive;
006import edu.wpi.first.wpilibj.can.CANTimeoutException;
007import edu.wpi.first.wpilibj.command.Subsystem;
008import edu.wpi.first.wpilibj.templates.RobotMap;
009import edu.wpi.first.wpilibj.templates.commands.ChassisDriveWithJoystick;
010import edu.wpi.first.wpilibj.templates.commands.CommandBase;
011
012public class Chassis extends Subsystem {
013  
014    CANJaguar leftMotor;
015    CANJaguar rightMotor;
016    RobotDrive drive;
017    boolean stopDriving;
018    
019    public Chassis() {
020        try {
021            leftMotor = new CANJaguar(RobotMap.leftMotor);
022            rightMotor = new CANJaguar(RobotMap.rightMotor);
023            leftMotor.changeControlMode(CANJaguar.ControlMode.kVoltage);
024            rightMotor.changeControlMode(CANJaguar.ControlMode.kVoltage);
025            //leftMotor.configEncoderCodesPerRev(360);
026            //leftMotor.setPositionReference(CANJaguar.PositionReference.kQuadEncoder);        
027            //rightMotor.configEncoderCodesPerRev(360);
028            //rightMotor.setPositionReference(CANJaguar.PositionReference.kQuadEncoder);
029            drive = new RobotDrive(leftMotor, rightMotor);   
030            stopDriving = true;
031        }
032        catch (CANTimeoutException ex) {
033            ;
034        }
035    }
036    public void initDefaultCommand() {
037        setDefaultCommand(new ChassisDriveWithJoystick());
038    }
039    public void DriveWithJoystick() {
040        Joystick stick = CommandBase.oi.getDriverJoystick();
041        double leftValue = stick.getRawAxis(2);  // opposite direction as calling rawAxis in tankDrive, 
042        double rightValue = stick.getRawAxis(5); //because tankDrive auto compensates for motors facing different directions
043        drive.tankDrive(leftValue, rightValue);
044    }    
045}
