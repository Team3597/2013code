006 * the wiring easier and significantly reduces the number of magic numbers
007 * floating around.
008 */
009public class RobotMap {
010    // For example to map the left and right motors, you could define the
011    // following variables to use with your drivetrain subsystem.
012    
013     public static final int loadMotor = 2;         // CAN ID for Jag
014     public static final int launchMotor = 3;       // CAN ID for Jag 
015     public static final int leftMotor = 5;         // CAN ID for Jag
016     public static final int rightMotor = 6;        // CAN ID for Jag
017     public static final int aimingMotor = 8;       // CAN ID for Jag                   
018}
